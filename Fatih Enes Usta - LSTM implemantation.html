def wordpieces(x):
    willCleaned = tokenizer_bert.tokenize(x)
    clean = []
    tokens = [",","","?","(",")","'","*","!"]
    for element in willCleaned:
        if element not in tokens:
            if not "#" in element:
                clean.append(element)
    return clean
